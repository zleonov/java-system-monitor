001package software.leonov.system.monitor;
002
003/**
004 * The system physical memory (RAM) usage metrics.
005 * 
006 * @author Zhenya Leonov
007 */
008public interface MemoryUsage {
009
010    /**
011     * Returns the amount of heap memory (in bytes) currently used by the JVM or -1 if the information is unavailable.
012     *
013     * @return the amount of heap memory (in bytes) currently used by the JVM or -1 if the information is unavailable
014     */
015    public long getUsedMemory();
016
017    /**
018     * Returns the amount of heap memory (in bytes) that is currently committed for the JVM's use or -1 if the information
019     * is unavailable. This is the amount of memory guaranteed to be available for the heap, which may be less than the
020     * maximum but more than the currently used memory.
021     *
022     * @return the amount of heap memory (in bytes) that is currently committed for the JVM's use or -1 if the information
023     *         is unavailable
024     */
025    public long getTotalMemory();
026
027    /**
028     * Returns the maximum observed amount of heap memory (in bytes) that has been used by the JVM or -1 if the information
029     * is unavailable.
030     *
031     * @return the maximum observed amount of heap memory (in bytes) that has been used by the JVM or -1 if the information
032     *         is unavailable
033     */
034    public long getMaxUsedMemory();
035
036}
