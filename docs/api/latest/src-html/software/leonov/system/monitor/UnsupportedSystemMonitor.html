001package software.leonov.system.monitor;
002
003/**
004 * A {@link SystemMonitor} that does nothing. All metrics return -1.
005 */
006public final class UnsupportedSystemMonitor implements SystemMonitor {
007
008    private static UnsupportedSystemMonitor INSTANCE = new UnsupportedSystemMonitor();
009
010    final static CpuUsage    NEGATIVE_CPU_USAGE    = new CpuUsageImpl(-1, -1, -1, -1, -1, -1, -1);
011    final static MemoryUsage NEGATIVE_MEMORY_USAGE = new MemoryUsageImpl(-1, -1, -1);
012
013    /**
014     * Returns a singleton instance of {@link UnsupportedSystemMonitor}.
015     * 
016     * @return a singleton instance of {@link UnsupportedSystemMonitor}
017     */
018    public static UnsupportedSystemMonitor getInstance() {
019        return INSTANCE;
020    }
021
022    private UnsupportedSystemMonitor() {
023    }
024
025    @Override
026    public void close() throws Exception {
027    }
028
029    @Override
030    public CpuUsage cpuUsage() {
031        return NEGATIVE_CPU_USAGE;
032    }
033
034    @Override
035    public MemoryUsage memoryUsage() {
036        return NEGATIVE_MEMORY_USAGE;
037    }
038
039}
