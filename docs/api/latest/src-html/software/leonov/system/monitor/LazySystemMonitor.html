001package software.leonov.system.monitor;
002
003import static java.util.Objects.requireNonNull;
004
005import java.time.Duration;
006
007/**
008 * A thread-safe {@link SystemMonitor} implementation that updates usage metrics on demand.
009 * <p>
010 * Usage metrics are updated when either {@link #getCpuUsage()} or {@link #getMemoryUsage()} is called, but only if a
011 * update threshold has elapsed since the last update. A {@link #withUpdateThreshold(Duration) custom} threshold can be
012 * specified when this class is created.
013 * <p>
014 * No underlying resources are managed by this monitor. The {@link #close()} method is a no-op and can be safely
015 * ignored.
016 *
017 * @author Zhenya Leonov
018 */
019public final class LazySystemMonitor extends AbstractSystemMonitor {
020
021    private static final Duration DEFAULT_UPDATE_THRESHOLD = Duration.ofMillis(250);
022
023    private final long updateThresholdMillis;
024    private long       lastUpdateTimeMillis = -1;
025
026    LazySystemMonitor() {
027        this(DEFAULT_UPDATE_THRESHOLD);
028    }
029
030    LazySystemMonitor(final Duration updateThreshold) {
031        this.updateThresholdMillis = updateThreshold.toMillis();
032    }
033
034    /**
035     * Creates a new {@link LazySystemMonitor} configured with the default update threshold.
036     * 
037     * @return a new {@link LazySystemMonitor} configured with the default update threshold
038     */
039    public static LazySystemMonitor withDefaultUpdateThreshold() {
040        return new LazySystemMonitor();
041    }
042
043    /**
044     * Creates a new {@link LazySystemMonitor} configured with the specified update threshold.
045     * <p>
046     * Usage metrics are updated when either {@link #getCpuUsage()} or {@link #getMemoryUsage()} is called, unless the
047     * specified {@code updateThreshold} time has not elapsed since the last update, in which case the most recently cached
048     * values are returned.
049     *
050     * @param updateThreshold the minimum time interval that must elapse between updates
051     * @return a new {@link LazySystemMonitor} configured with the specified update threshold
052     */
053    public static LazySystemMonitor withUpdateThreshold(final Duration updateThreshold) {
054        requireNonNull(updateThreshold, "updateThreshold == null");
055        if (updateThreshold.isNegative() || updateThreshold.isZero())
056            throw new IllegalArgumentException("updateThreshold <= 0");
057        return new LazySystemMonitor(updateThreshold);
058    }
059
060    @Override
061    public CpuUsage getCpuUsage() {
062        updateMetrics();
063        return super.getCpuUsage();
064    }
065
066    @Override
067    public MemoryUsage getMemoryUsage() {
068        updateMetrics();
069        return super.getMemoryUsage();
070    }
071
072    @Override
073    protected synchronized void updateMetrics() {
074        final long currentTimeMillis = System.currentTimeMillis();
075
076        if (lastUpdateTimeMillis == -1 || (currentTimeMillis - lastUpdateTimeMillis) >= updateThresholdMillis) {
077            super.updateMetrics();
078            lastUpdateTimeMillis = currentTimeMillis;
079        }
080    }
081
082}
