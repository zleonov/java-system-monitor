001package software.leonov.system.monitor;
002
003import static java.util.Objects.requireNonNull;
004
005import java.time.Duration;
006
007/**
008 * A thread-safe {@link SystemMonitor} implementation that refreshes usage metrics on demand.
009 * <p>
010 * This monitor employs a debouncing mechanism to prevent excessive resource consumption. Usage metrics are updated when
011 * either {@link #getCpuUsage()} or {@link #getMemoryUsage()} is called, but only if a refresh threshold has elapsed
012 * since the last update. A {@link #withRefreshThreshold(Duration) custom} threshold can be specified when this class is
013 * created.
014 * <p>
015 * No underlying resources are managed by this monitor. The {@link #close()} method is a no-op and can be safely
016 * ignored.
017 *
018 * @author Zhenya Leonov
019 */
020public final class LazySystemMonitor extends AbstractSystemMonitor {
021
022    private static final Duration DEFAULT_REFRESH_THRESHOLD = Duration.ofMillis(250);
023
024    private final long refreshThresholdMillis;
025    private long       lastRefreshTimeMillis = -1;
026
027    LazySystemMonitor() {
028        this(DEFAULT_REFRESH_THRESHOLD);
029    }
030
031    LazySystemMonitor(final Duration refreshThreshold) {
032        this.refreshThresholdMillis = refreshThreshold.toMillis();
033    }
034
035    /**
036     * Creates a new {@link LazySystemMonitor} configured with the default refresh threshold.
037     * 
038     * @return a new {@link LazySystemMonitor} configured with the default refresh threshold
039     */
040    public static LazySystemMonitor withDefaultRefreshThreshold() {
041        return new LazySystemMonitor();
042    }
043
044    /**
045     * Creates a new {@link LazySystemMonitor} configured with the specified refresh threshold.
046     * <p>
047     * Usage metrics are updated when either {@link #getCpuUsage()} or {@link #getMemoryUsage()} is called, unless the
048     * specified {@code refreshThreshold} time has not elapsed since the last update, in which case the most recently cached
049     * values are returned.
050     *
051     * @param refreshThreshold the minimum time interval that must elapse between updates
052     * @return a new {@link LazySystemMonitor} configured with the specified refresh threshold
053     */
054    public static LazySystemMonitor withRefreshThreshold(final Duration refreshThreshold) {
055        requireNonNull(refreshThreshold, "refreshThreshold == null");
056        if (refreshThreshold.isNegative() || refreshThreshold.isZero())
057            throw new IllegalArgumentException("refreshThreshold <= 0");
058        return new LazySystemMonitor(refreshThreshold);
059    }
060
061    @Override
062    public CpuUsage getCpuUsage() {
063        refreshMetrics();
064        return super.getCpuUsage();
065    }
066
067    @Override
068    public MemoryUsage getMemoryUsage() {
069        refreshMetrics();
070        return super.getMemoryUsage();
071    }
072
073    @Override
074    protected synchronized void refreshMetrics() {
075        final long currentTimeMillis = System.currentTimeMillis();
076
077        if (lastRefreshTimeMillis == -1 || (currentTimeMillis - lastRefreshTimeMillis) >= refreshThresholdMillis) {
078            super.refreshMetrics();
079            lastRefreshTimeMillis = currentTimeMillis;
080        }
081    }
082
083}
