001package software.leonov.system.monitor;
002
003import static java.util.Objects.requireNonNull;
004
005import java.time.Duration;
006
007/**
008 * A {@link SystemMonitor} that refreshes usage metrics on demand.
009 * <p>
010 * This class is created with a {@link #withDefaultRefreshThreshold() default} or a
011 * {@link #withRefreshThreshold(Duration) specified} refresh threshold. Usage metrics are updated when either
012 * {@link #cpuUsage()} or {@link #memoryUsage()} is called, unless refresh threshold time has not elapsed since the last
013 * update, in which case the most recently cached values are returned.
014 * <p>
015 * This class does not make use of any underlying closeable resources; it's {@link #close()} method is a no-op.
016 * 
017 * @author Zhenya Leonov
018 */
019
020/**
021 * A thread-safe {@link SystemMonitor} implementation that refreshes usage metrics on demand.
022 * <p>
023 * This monitor employs a debouncing mechanism to prevent excessive resource consumption. Usage metrics are updated when
024 * either {@link #getCpuUsage()} or {@link #getMemoryUsage()} is called, but only if a refresh threshold has elapsed since the
025 * last update. A {@link #withRefreshThreshold(Duration) custom} threshold can be specified when this class is created.
026 * <p>
027 * No underlying resources are managed by this monitor. The {@link #close()} method is a no-op and can be safely
028 * ignored.
029 *
030 * @author Zhenya Leonov
031 */
032public final class LazySystemMonitor extends AbstractSystemMonitor {
033
034    private static final Duration DEFAULT_REFRESH_THRESHOLD = Duration.ofMillis(250);
035
036    private final long    refreshThresholdMillis;
037    private volatile long lastRefreshTimeMillis = -1;
038
039    LazySystemMonitor() {
040        this(DEFAULT_REFRESH_THRESHOLD);
041    }
042
043    LazySystemMonitor(final Duration refreshThreshold) {
044        this.refreshThresholdMillis = refreshThreshold.toMillis();
045    }
046
047    /**
048     * Creates a new {@link LazySystemMonitor} configured with the default refresh threshold.
049     * 
050     * @return a new {@link LazySystemMonitor} configured with the default refresh threshold
051     */
052    public static LazySystemMonitor withDefaultRefreshThreshold() {
053        return new LazySystemMonitor();
054    }
055
056    /**
057     * Creates a new {@link LazySystemMonitor} configured with the specified refresh threshold.
058     * <p>
059     * Usage metrics are updated when either {@link #getCpuUsage()} or {@link #getMemoryUsage()} is called, unless the specified
060     * {@code refreshThreshold} time has not elapsed since the last update, in which case the most recently cached values
061     * are returned.
062     *
063     * @param refreshThreshold the minimum time interval that must elapse between updates
064     * @return a new {@link LazySystemMonitor} configured with the specified refresh threshold
065     */
066    public static LazySystemMonitor withRefreshThreshold(final Duration refreshThreshold) {
067        requireNonNull(refreshThreshold, "refreshThreshold == null");
068        if (refreshThreshold.isNegative() || refreshThreshold.isZero())
069            throw new IllegalArgumentException("refreshThreshold <= 0");
070        return new LazySystemMonitor(refreshThreshold);
071    }
072
073    @Override
074    public CpuUsage getCpuUsage() {
075        refreshMetrics();
076        return super.getCpuUsage();
077    }
078
079    @Override
080    public MemoryUsage getMemoryUsage() {
081        refreshMetrics();
082        return super.getMemoryUsage();
083    }
084
085    @Override
086    protected synchronized void refreshMetrics() {
087        final long currentTimeMillis = System.currentTimeMillis();
088
089        if (lastRefreshTimeMillis == -1 || (currentTimeMillis - lastRefreshTimeMillis) >= refreshThresholdMillis) {
090            super.refreshMetrics();
091            lastRefreshTimeMillis = currentTimeMillis;
092        }
093    }
094
095    @Override
096    public void close() throws Exception {
097        // nothing to do
098    }
099
100}
