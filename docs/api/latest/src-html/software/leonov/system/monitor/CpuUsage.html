001package software.leonov.system.monitor;
002
003import java.lang.management.OperatingSystemMXBean;
004
005/**
006 * The Central Processing Unit (CPU) usage metrics.
007 * 
008 * @author Zhenya Leonov
009 */
010public interface CpuUsage {
011
012    /**
013     * Returns the JVM process CPU usage or -1.0 if it is not supported.
014     * 
015     * @return the JVM process CPU usage or -1.0 if it is not supported
016     */
017    public double getProcessCpuLoad();
018
019    /**
020     * Returns the system-wide CPU usage or -1.0 if it is not supported.
021     * 
022     * @return the system-wide CPU usage or -1.0 if it is not supported
023     */
024    public double getSystemCpuLoad();
025
026    /**
027     * Returns the system load average for the last minute (only supported on Unix/Linux systems) or -1.0 if it is not
028     * supported.
029     * <p>
030     * For more details see {@link OperatingSystemMXBean#getSystemLoadAverage()}.
031     * 
032     * @return the system load average for the last minute or -1.0 if it is not supported
033     */
034    public double getSystemLoadAverage();
035
036    /**
037     * Returns the average JVM process CPU usage or -1.0 if it is not supported or not ready.
038     * 
039     * @return the average JVM process CPU usage or -1.0 if it is not supported or not ready
040     */
041    public double getAverageProcessCpuLoad();
042
043    /**
044     * Returns average system-wide CPU usage or -1.0 if it is not supported or not ready.
045     * 
046     * @return average system-wide CPU usage or -1.0 if it is not supported or not ready
047     */
048    public double getAverageSystemCpuLoad();
049
050    /**
051     * Returns the maximum observed JVM process CPU usage (0.0 to 100.0) or -1.0 if it is not supported or not ready.
052     *
053     * @return the maximum observed JVM process CPU usage (0.0 to 100.0) or -1.0 if it is not supported or not ready
054     */
055    public double getMaxProcessCpuLoad();
056
057    /**
058     * Returns the maximum observed system-wide CPU usage (0.0 to 100.0) or -1.0 if it is not supported or not ready.
059     *
060     * @return the maximum observed system-wide CPU usage (0.0 to 100.0) or -1.0 if it is not supported or not ready
061     */
062    public double getMaxSystemCpuLoad();
063
064}
