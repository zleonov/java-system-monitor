001package software.leonov.system.monitor.util;
002
003/**
004 * Basic utility methods for formatting CPU and memory usage metrics.
005 * <p>
006 * This is a convenience class intended for casual use only (primarily for quick testing and debugging).
007 * 
008 * @author Zhenya Leonov
009 */
010public final class Formatter {
011
012    private static final String PRE          = "KMGTPE";
013    private static final int    DECIMAL_UNIT = 1000;
014    private static final int    BINARY_UNIT  = 1024;
015
016    private Formatter() {
017    }
018
019    /**
020     * Formats the given percent value into a human-readable string.
021     * 
022     * @param pct the percent value to format
023     * @return a formatted string (e.g., 12.75%) or "N/A" if the value is negative
024     */
025    public static String formatPercent(final double pct) {
026        return pct < 0 ? "N/A" : String.format("%.2f%%", pct);
027    }
028
029    /**
030     * Formats the given number of bytes into a human-readable string in the binary (1024) base unit.
031     *
032     * @param bytes the number of bytes to format
033     * @return a formatted string (e.g., "1.23 MiB") or "N/A" if the value is negative
034     */
035    public static String formatBinaryBytes(final long bytes) {
036        return formatBytes(bytes, BINARY_UNIT);
037    }
038
039    /**
040     * Formats the given number of bytes into a human-readable string in the decimal (1000) base unit.
041     *
042     * @param bytes the number of bytes to format
043     * @return a formatted string (e.g., "1.23 MB") or "N/A" if the value is negative
044     */
045    public static String formatDecimalBytes(final long bytes) {
046        return formatBytes(bytes, DECIMAL_UNIT);
047    }
048
049    private static String formatBytes(final long bytes, final int base) {
050        if (bytes < 0)
051            return "N/A";
052
053        if (bytes < base)
054            return bytes + " bytes";
055
056        final int    exp     = (int) (Math.log(bytes) / Math.log(base));
057        final char   pre     = PRE.charAt(exp - 1);
058        final String pattern = base == BINARY_UNIT ? "%.2f%siB" : "%.2f%sB";
059
060        return String.format(pattern, bytes / Math.pow(base, exp), pre);
061    }
062
063}
